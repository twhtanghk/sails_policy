addFile = (cfg, file) ->
  ret = {}
  ret[path.parse(file).name] = require file
  _.defaults cfg, ret

addPattern = (cfg, pattern) ->
  glob
    .sync pattern
    .reduce addFile, cfg

module.exports =
  _.reduce list, addPattern, {}
